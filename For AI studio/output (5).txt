    <script>
        function showTab(tabName) {
            // Hide all tabs
            document.getElementById('dashboard-tab').classList.add('hidden');
            document.getElementById('request-tab').classList.add('hidden');
            document.getElementById('payment-tab').classList.add('hidden');
            document.getElementById('supplementary-tab').classList.add('hidden');
            document.getElementById('history-tab').classList.add('hidden');
            
            // Show selected tab
            document.getElementById(tabName + '-tab').classList.remove('hidden');
            
            // Update active tab
            document.querySelectorAll('.tab').forEach(tab => {
                tab.classList.remove('active');
            });
            event.target.classList.add('active');
        }
        
        function submitRequest() {
            alert('درخواست شما با موفقیت ثبت گردید. درخواست در حال بررسی توسط مربی می‌باشد. لطفا شکیبا باشید.');
            showTab('dashboard');
        }
        
        function handleFileSelect(file) {
            if (file) {
                const fileSize = file.size / 1024; // KB
                if (fileSize > 500) {
                    alert('حجم عکس آپلود شده بیش از 500 کیلوبایت است، لطفا حجم عکس را کاهش دهید.');
                    return;
                }
                
                document.getElementById('file-name').textContent = 'نام فایل: ' + file.name;
                document.getElementById('file-size').textContent = 'حجم فایل: ' + fileSize.toFixed(2) + ' KB';
                
                const reader = new FileReader();
                reader.onload = function(e) {
                    document.getElementById('preview-image').src = e.target.result;
                }
                reader.readAsDataURL(file);
                
                document.getElementById('file-preview').classList.remove('hidden');
                document.getElementById('submit-payment-btn').disabled = false;
            }
        }
        
        function submitPayment() {
            alert('درخواست شما با موفقیت ثبت گردید. برنامه شما پس از تأیید ۲ تا ۴ روز کاری توسط مربی برای شما ارسال خواهد شد و شما می‌توانید در پنل خود آنرا مشاهده کنید.');
            showTab('supplementary');
        }
        
        function handlePhotoUpload(input, previewId) {
            if (input.files && input.files[0]) {
                const file = input.files[0];
                const fileSize = file.size / 1024; // KB
                if (fileSize > 500) {
                    alert('حجم عکس آپلود شده بیش از 500 کیلوبایت است، لطفا حجم عکس را کاهش دهید.');
                    input.value = '';
                    return;
                }
                
                const reader = new FileReader();
                reader.onload = function(e) {
                    document.getElementById(previewId).src = e.target.result;
                }
                reader.readAsDataURL(file);
            }
        }
        
        function submitSupplementaryInfo() {
            alert('فرم با موفقیت ثبت شد. منتظر ارسال برنامه از جانب مربی باشید.');
            showTab('dashboard');
        }
        
        // Show payment tab when notification is clicked
        document.addEventListener('DOMContentLoaded', function() {
            // Simulate notification click
            setTimeout(() => {
                const notification = document.querySelector('.notification');
                if (notification) {
                    notification.addEventListener('click', () => {
                        showTab('payment');
                    });
                }
            }, 1000);
        });
    </script>
